378 def _wait_delay_check(self, func_to_check, millis, iterations, delta):
379 """ "
380 call func_to_check(millis) "iterations" times and check each time if
381 function "waited" for given millisecond (+- delta). At the end, take
382 average time for each call (whole_duration/iterations), which should
383 be equal to millis (+- delta - acceptable margin of error).
384 *Created to avoid code duplication during delay and wait tests
385 """
386 # take starting time for duration calculation
387 start_time = time.time()
388 for i in range(iterations):
389 wait_time = func_to_check(millis)
390 # Check equality of wait_time and millis with margin of error delta
391 self.assertAlmostEqual(wait_time, millis, delta=delta)
392 stop_time = time.time()
393 # Cycle duration in millisecond
394 duration = round((stop_time - start_time) * 1000)
395 # Duration/Iterations should be (almost) equal to predefined millis
396 self.assertAlmostEqual(duration / iterations, millis, delta=delta)
