69 @pytest.mark.parametrize("dtype", np.typecodes["AllFloat"])
70 def test_nep50_weak_integers_with_inexact(dtype):
71 # Avoids warning (different code path for scalars)
72 scalar_type = np.dtype(dtype).type
73  
74 too_big_int = int(np.finfo(dtype).max) * 2
75  
76 if dtype in "dDG":
77 # These dtypes currently convert to Python float internally, which
78 # raises an OverflowError, while the other dtypes overflow to inf.
79 # NOTE: It may make sense to normalize the behavior!
80 with pytest.raises(OverflowError):
81 scalar_type(1) + too_big_int
82  
83 with pytest.raises(OverflowError):
84 np.array(1, dtype=dtype) + too_big_int
85 else:
86 # NumPy uses (or used) `int -> string -> longdouble` for the
87 # conversion. But Python may refuse `str(int)` for huge ints.
88 # In that case, RuntimeWarning would be correct, but conversion
89 # fails earlier (seems to happen on 32bit linux, possibly only debug).
90 if dtype in "gG":
91 try:
92 str(too_big_int)
93 except ValueError:
94 pytest.skip("`huge_int -> string -> longdouble` failed")
95  
96 # Otherwise, we overflow to infinity:
97 with pytest.warns(RuntimeWarning):
98 res = scalar_type(1) + too_big_int
99 assert res.dtype == dtype
100 assert res == np.inf

157 def test_nep50_huge_integers(ufunc):
158 # Very large integers are complicated, because they go to uint64 or
159 # object dtype. This tests covers a few possible paths.
160 with pytest.raises(OverflowError):
161 ufunc(np.int64(0), 2**63) # 2**63 too large for int64
162  
163 with pytest.raises(OverflowError):
164 ufunc(np.uint64(0), 2**64) # 2**64 cannot be represented by uint64
165  
166 # However, 2**63 can be represented by the uint64 (and that is used):
167 res = ufunc(np.uint64(1), 2**63)
168

171  
172 # The following paths fail to warn correctly about the change:
