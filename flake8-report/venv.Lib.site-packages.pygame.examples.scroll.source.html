2 """ pygame.examples.scroll
3  
4 An zoomed image viewer that demonstrates Surface.scroll
5  
6 This example shows a scrollable image that has a zoom factor of eight.
7 It uses the Surface.scroll function to shift the image on the display
8 surface. A clip rectangle protects a margin area. If called as a function,
9 the example accepts an optional image file path. If run as a program
10 it takes an optional file path command line argument. If no file
11 is provided a default image file is used.
12  
13 When running click on a black triangle to move one pixel in the direction
14 the triangle points. Or use the arrow keys. Close the window or press ESC
15 to quit.
16 """

52 def scroll_view(screen, image: pg.Surface, direction: int, view_rect):
53 src_rect = None
54 dst_rect = None
55 zoom_view_rect = screen.get_clip()
56 image_w, image_h = image.get_size()
57 if direction == DIR_UP:
58 if view_rect.top > 0:
59 screen.scroll(dy=zoom_factor)
60 view_rect.move_ip(0, -1)
61 src_rect = view_rect.copy()
62 src_rect.h = 1
63 dst_rect = zoom_view_rect.copy()
64 dst_rect.h = zoom_factor
65 elif direction == DIR_DOWN:
66 if view_rect.bottom < image_h:
67 screen.scroll(dy=-zoom_factor)
68 view_rect.move_ip(0, 1)
69 src_rect = view_rect.copy()
70 src_rect.h = 1
71 src_rect.bottom = view_rect.bottom
72 dst_rect = zoom_view_rect.copy()
73 dst_rect.h = zoom_factor
74 dst_rect.bottom = zoom_view_rect.bottom
75 elif direction == DIR_LEFT:
76 if view_rect.left > 0:
77 screen.scroll(dx=zoom_factor)
78 view_rect.move_ip(-1, 0)
79 src_rect = view_rect.copy()
80 src_rect.w = 1
81 dst_rect = zoom_view_rect.copy()
82 dst_rect.w = zoom_factor
83 elif direction == DIR_RIGHT:
84 if view_rect.right < image_w:
85 screen.scroll(dx=-zoom_factor)

88 src_rect.w = 1

99 def main(image_file=None):
100 if image_file is None:
101 image_file = os.path.join(main_dir, "data", "arraydemo.bmp")
102 margin = 80
103 view_size = (30, 20)
104 zoom_view_size = (view_size[0] * zoom_factor, view_size[1] * zoom_factor)
105 win_size = (zoom_view_size[0] + 2 * margin, zoom_view_size[1] + 2 * margin)
106 background_color = pg.Color("beige")
107  
108 pg.init()
109 pg.display.set_caption("Scroll Example")

111 # set up key repeating so we can hold down the key to scroll.
112 old_k_delay, old_k_interval = pg.key.get_repeat()

115 try:
116 screen = pg.display.set_mode(win_size)
117 screen.fill(background_color)
118 pg.display.flip()

122  
123 if image_w < view_size[0] or image_h < view_size[1]:

126 return
