249 def word_wrap(surf, text, font, margin=0, color=(0, 0, 0)):
250 font.origin = True
251 surf_width, surf_height = surf.get_size()
252 width = surf_width - 2 * margin
253 height = surf_height - 2 * margin
254 line_spacing = int(1.25 * font.get_sized_height())
255 x, y = margin, margin + line_spacing
256 space = font.get_rect(" ")
257 for word in iwords(text):
258 if word == "\n":
259 x, y = margin, y + line_spacing
260 else:
261 bounds = font.get_rect(word)
262 if x + bounds.width + bounds.x >= width:
263 x, y = margin, y + line_spacing
264 if x + bounds.width + bounds.x >= width:
265 raise ValueError("word too wide for the surface")
266 if y + bounds.height - bounds.y >= height:
267 raise ValueError("text to long for the surface")
268 font.render_to(surf, (x, y), None, color)
269 x += bounds.width + space.width
270 return x, y

273 def iwords(text):
274 # r"\n|[^ ]+"
275 #
276 head = 0
277 tail = head
278 end = len(text)
279 while head < end:
280 if text[head] == " ":
281 head += 1
282 tail = head + 1
283 elif text[head] == "\n":
284 head += 1
285 yield "\n"
286 tail = head + 1
287 elif tail == end:
288 yield text[head:]
289 head = end
290 elif text[tail] == "\n":
291 yield text[head:tail]
292 head = tail
293 elif text[tail] == " ":
294 yield text[head:tail]
295 head = tail
296 else:
297 tail += 1
