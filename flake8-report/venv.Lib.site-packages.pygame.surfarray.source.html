21 """pygame module for accessing surface pixel data using array interfaces
22  
23 Functions to convert between NumPy arrays and Surface objects. This module
24 will only be functional when pygame can use the external NumPy package.
25 If NumPy can't be imported, surfarray becomes a MissingModule object.
26  
27 Every pixel is stored as a single integer value to represent the red,
28 green, and blue colors. The 8bit images use a value that looks into a
29 colormap. Pixels with higher depth use a bit packing process to place
30 three or four values into a single number.
31  
32 The arrays are indexed by the X axis first, followed by the Y
33 axis. Arrays that treat the pixels as a single integer are referred to
34 as 2D arrays. This module can also separate the red, green, and blue
35 color values into separate indices. These types of arrays are referred
36 to as 3D arrays, and the last index is 0 for red, 1 for green, and 2 for
37 blue.
38 """

97 def blit_array(surface, array):
98 """pygame.surfarray.blit_array(Surface, array): return None
99  
100 Blit directly from a array values.
101  
102 Directly copy values from an array into a Surface. This is faster than
103 converting the array into a Surface and blitting. The array must be the
104 same dimensions as the Surface and will completely replace all pixel
105 values. Only integer, ascii character and record arrays are accepted.
106  
107 This function will temporarily lock the Surface as the new values are
108 copied.
109 """
110 if isinstance(array, numpy_ndarray) and array.dtype in numpy_floats:
111 array = array.round(0).astype(numpy_uint32)
112 return array_to_surface(surface, array)

128 def array2d(surface):
129 """pygame.surfarray.array2d(Surface): return array
130  
131 copy pixels into a 2d array
132  
133 Copy the pixels from a Surface into a 2D array. The bit depth of the
134 surface will control the size of the integer values, and will work
135 for any type of pixel format.
136  
137 This function will temporarily lock the Surface as pixels are copied
138 (see the Surface.lock - lock the Surface memory for pixel access
139 method).
140 """
141 bpp = surface.get_bytesize()
142 try:
143 dtype = (numpy.uint8, numpy.uint16, numpy.int32, numpy.int32)[bpp - 1]
144 except IndexError:
145 raise ValueError(f"unsupported bit depth {bpp * 8} for 2D array")
146 size = surface.get_size()
147 array = numpy.empty(size, dtype)
148 surface_to_array(array, surface)
149 return array

380 def map_array(surface, array):
381 """pygame.surfarray.map_array(Surface, array3d): return array2d
382  
383 map a 3d array into a 2d array
384  
385 Convert a 3D array into a 2D array. This will use the given Surface
386 format to control the conversion.
387  
388 Note: arrays do not need to be 3D, as long as the minor axis has
389 three elements giving the component colours, any array shape can be
390 used (for example, a single colour can be mapped, or an array of
391 colours). The array shape is limited to eleven dimensions maximum,
392 including the three element minor axis.
393 """
394 if array.ndim == 0:
395 raise ValueError("array must have at least 1 dimension")
396 shape = array.shape
397 if shape[-1] != 3:
398 raise ValueError("array must be a 3d array of 3-value color data")
399 target = numpy_empty(shape[:-1], numpy.int32)
400 pix_map_array(target, array, surface)
401 return target
